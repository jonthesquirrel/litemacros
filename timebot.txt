IF(@#timebot_time > 0)
    #m = ((@#timebot_time - (@#timebot_time-(60*(@#timebot_time/60)))) / 60)
    #s = (@#timebot_time-(60*(@#timebot_time/60)))
    &human_time = ""
    IF(#m != 0)
        &human_time = "%&human_time%%#m%m"
    ENDIF
    IF(#s != 0)
        &human_time = "%&human_time%%#s%s"
    ENDIF
ELSE
    &human_time = "0s"
ENDIF

&arg_mode = $$[1]
IF(&arg_mode = "set")
    UNSET(&arg_time)
    UNSET(@#timebot_time)
    UNSET(#min)
    UNSET(#sec)
    &arg_time = $$[2]
    MATCH(%&arg_time%,"(\d+)m",#min,1)
    MATCH(%&arg_time%,"(\d+)s",#sec,1)
    @#timebot_time = (#min * 60) + #sec
    LOG("<timebot> started for %&arg_time% (%@#timebot_time%)")
    //start FOR loop counting down
    //timer finishing logic
    IF(@&timebot_cmd);ECHO(@&timebot_cmd);ENDIF
ELSEIF(&arg_mode = "stop")
    UNSET(@#timebot_time)
    LOG("<timebot> stopped")
ELSEIF(&arg_mode = "add")
    UNSET(&arg_time)
    &arg_time = $$[2]
    MATCH(%&arg_time%,"(\d+)m",#min,1)
    MATCH(%&arg_time%,"(\d+)s",#sec,1)
    #temp_time = (#min * 60) + #sec
    INC(@#timebot_time,%#temp_time%)
    LOG("<timebot> added %&arg_time% to %&human_time%")
ELSEIF(&arg_mode = "rem")
    UNSET(&arg_time)
    &arg_time = $$[2]
    MATCH(%&arg_time%,"(\d+)m",#min,1)
    MATCH(%&arg_time%,"(\d+)s",#sec,1)
    #temp_time = (#min * 60) + #sec
    IF(#temp_time < @#timebot_time)
        DEC(@#timebot_time,%#temp_time%)
    ELSE
        @#timebot_time = 0
    ENDIF
    LOG("<timebot> removed %&arg_time% from %&human_time%")
ELSEIF(&arg_mode = "end")
    @#timebot_time = 0
ELSEIF(&arg_mode = "say")
    ECHO("<timebot> %&human_time%")
ELSEIF(&arg_mode = "get")
    LOG("<timebot> %&human_time%")
ELSEIF(&arg_mode = "cmd")
    @&timebot_cmd = $$[2]
ELSE
    LOG("<timebot> usage: /!timer <set <Xm|Xs|XmXs>|cancel|end|add|rem|say|get|cmd [command for end]>")
    //TODO: pause|resume
ENDIF
